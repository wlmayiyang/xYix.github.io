int N, S;
int X[100005]; ll A[100005];
ll ANS;
int Solve(int l, int r) {
    if(S < X[l]) return ANS += X[r] - S, r;
    if(S > X[r]) return ANS += S - X[l], l;
    if(A[l] >= A[r]) {
        A[l] += A[r];
        ANS += X[r] - X[Solve(l, r - 1)];
        return r;
    }
    else {
        A[r] += A[l];
        ANS += X[Solve(l + 1, r)] - X[l];
        return l;
    }
}
