int n, p;
int pf[55][2505];
void get_pf() {
    int a[55];
    for (int i = 0; i <= n; i++) scanf("%d", &a[i]);
    pf[0][0] = 1;
    for (int i = 1; i < p; i++)
    for (int j = 0; j <= i * n; j++) {
        for (int k = 0; k <= n && k <= j; k++)
            pf[i][j] += a[k] * pf[i - 1][j - k];
        pf[i][j] %= p;
    }
}

vector<int> solve(ll m, ll k) {
    // (m, k), ..., (m, k - 2n + 1)
    vector<int> ans; ans.resize(n + n);
    int m0 = m % p;
    if (m < p) {
        for (int i = 0; i < n + n; i++)
            if (k - i >= 0 && k - i <= m0 * n) ans[i] = pf[m0][(int)(k - i)];
            else ans[i] = 0;
        return ans;
    }
    vector<int> lst = solve(m / p, k / p);
    for (ll i = k; i > k - n - n && i >= 0; i--) {
        ll j0 = i / p, j1 = i - p * j0;
        while (1) {
            if (j1 > m0 * n) break;
            assert((int)(k / p - j0) < n + n);
            int tg = lst[(int)(k / p - j0)];
            int tf = pf[m0][j1];
            ans[(int)(k - i)] += tg * tf;
            j0--, j1 += p;
        }
        ans[(int)(k - i)] %= p;
    }
    return ans;
}
