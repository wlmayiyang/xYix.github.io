struct leftist{
    int c[2];
    int dis, val;
    int siz;
} H[100005];
void pushup(int x) {
    if (H[H[x].c[1]].dis < H[H[x].c[0]].dis) swap(H[x].c[0], H[x].c[1]);
    H[x].dis = H[H[x].c[0]].dis + 1;
    H[x].siz = H[H[x].c[0]].siz + H[H[x].c[1]].siz + 1;
}
int merge(int x, int y) {
    if (!x || !y) return x + y;
    if (H[x].val < H[y].val) swap(x, y);
    H[x].c[0] = merge(H[x].c[0], y);
    pushup(x);
    return x;
}
