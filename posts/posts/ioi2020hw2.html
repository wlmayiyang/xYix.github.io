AGC034E - 完全压缩

首先枚举最终目标节点，以其为根。然后盲猜一手：不会操作两个有祖先关系的点。最后从把所有点的深度消成 0 出发，编出一个树形 DP 就做完了。

果然 ATC/CF 就是敢猜才会赢！

int f[2005], siz[2005], sum[2005];
void solve(int x, int fa) {
    siz[x] = clr[x];
    sum[x] = 0;
    int son = 0;
    for (int y : G[x]) if (y != fa) {
        solve(y, x);
        siz[x] += siz[y];
        sum[x] += (sum[y] += siz[y]);
        if (sum[y] > sum[son]) son = y;
    }
    if (!son) { f[x] = 0; return; }
    if (2 * sum[son] <= sum[x]) f[x] = sum[x] / 2;
    else f[x] = sum[x] - sum[son] + min(2 * f[son], 2 * sum[son] - sum[x]) / 2;
}
